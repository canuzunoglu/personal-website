```bash
# Installation with `git clone`
git clone my-site https://github.com/canuzunoglu/personal-website.git
cd personal-website
yarn install

# To develop
yarn develop

# To build
yarn build

# To test SSR (for Lighthouse etc.)
yarn ssr

# To format JS (precommit)
yarn format

# To generate favicons (included in `build`)
yarn build:favicons
```
